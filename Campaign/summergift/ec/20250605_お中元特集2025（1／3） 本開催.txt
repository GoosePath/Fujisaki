<script>
  // MV内の部分リンク設定
  var setScrollbarWidth = function () {
    var scrollbarWidth = window.innerWidth - document.documentElement.clientWidth;
    var pcBreakPoint = 1100;
    var spBreakPoint = 375;
    document.documentElement.style.setProperty('--scrollbar-width', scrollbarWidth);
    document.documentElement.style.setProperty('--pc-break-point', pcBreakPoint);
    document.documentElement.style.setProperty('--sp-break-point', spBreakPoint);
  };

  window.addEventListener('DOMContentLoaded', setScrollbarWidth);
  window.addEventListener('resize', setScrollbarWidth);

  $(function () {
    $('.mainvisual-link').on('click', function () {
      location.href = $(this).data('href');
    });

    var slideHead = $('.js-slideHead'),
      slideContent = $('.js-slideContent');

    slideContent.css('display', 'none');

    slideHead.on('click', function (e) {
      var $this = $(this);
      var index = slideHead.index($this);
      var currentContent = slideContent.eq(index);

      $this.toggleClass('open');

      if ($this.hasClass('open')) {
        currentContent.stop().slideDown(200);
      } else {
        currentContent.stop().slideUp(200);
      }

      if ($this.children('span').text() === '詳細を見る') {
        $this.children('span').text('閉じる');
      } else {
        if ($this.children('span').text() === '閉じる') {
          $this.children('span').text('詳細を見る');
        }
      }
    });

    // ランキング
    if ($('.summergiftRankingSwiper').length) {
      var summergiftRankingSwiper = new Swiper(".summergiftRankingSwiper", {
        slidesPerView: 5,
        slidesPerGroup: 1,
        spaceBetween: 13,
        loop: false,
        allowTouchMove: false,
        navigation: {
          nextEl: ".summergiftRankingSwiperSlideNext",
          prevEl: ".summergiftRankingSwiperSlidePrev",
        },
        on: {
          init: function () {
            var imgHeight = $('.summergiftRankingSwiper .swiper-slide img').eq(0).innerWidth();
            $('.summergiftRankingSwiperSlideNext').css('top', (imgHeight / 2) + 'px');
            $('.summergiftRankingSwiperSlidePrev').css('top', (imgHeight / 2) + 'px');
          },
          resize: function () {
            var imgHeight = $('.summergiftRankingSwiper .swiper-slide img').eq(0).innerWidth();
            $('.summergiftRankingSwiperSlideNext').css('top', (imgHeight / 2) + 'px');
            $('.summergiftRankingSwiperSlidePrev').css('top', (imgHeight / 2) + 'px');
          },
        },
        breakpoints: {
          767: {
            loop: true,
            allowTouchMove: true,
            slidesPerView: 2,
            slidesPerGroup: 1,
            spaceBetween: 15,
          },
        }
      });

      $('.summergiftRankingSwiper .item__name').matchHeight();
    }
  });
</script>
